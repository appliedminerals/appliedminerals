        // Function to switch between pages
        function showPage(pageId) {
            const pages = document.querySelectorAll('.page-content');
            pages.forEach(page => page.classList.add('hidden'));

            const activePage = document.getElementById(`${pageId}-page`);
            if (activePage) {
                activePage.classList.remove('hidden');
            }
        }

        // Initialize to Home Page
        document.addEventListener('DOMContentLoaded', () => {
            showPage('home');

            const form = document.getElementById('inquiry-form');
            const successMessage = document.getElementById('success-message');

            form.addEventListener('submit', function(event) {
                event.preventDefault();

                // 1. Collect form data
                const name = document.getElementById('name').value;
                const phone = document.getElementById('phone').value;
                const email = document.getElementById('email').value;
                const message = document.getElementById('message').value;

                // 2. Construct the email link (mailto)
                // THE RECIPIENT REMAINS HARDCODED IN THE JAVASCRIPT
                const recipient = 'ccarney@appliedminerals.com';
                const subject = encodeURIComponent('New Website Inquiry from ' + name);
                
                let body = `Name: ${name}\n`;
                body += `Email: ${email}\n`;
                if (phone) body += `Phone: ${phone}\n\n`;
                body += `--- Inquiry ---\n${message}`;

                const mailtoLink = `mailto:${recipient}?subject=${subject}&body=${encodeURIComponent(body)}`;

                // 3. Open mail client (This is a mock interaction since we can't send real emails)
                // window.location.href = mailtoLink; 
                
                // Show success message and clear form
                successMessage.classList.remove('hidden');
                form.reset();
                
                // Hide success message after 5 seconds
                setTimeout(() => {
                    successMessage.classList.add('hidden');
                }, 5000);

                console.log(`Email prepared for: ${recipient}`);
                console.log(`Subject: ${decodeURIComponent(subject)}`);
                console.log(`Body: ${decodeURIComponent(body)}`);
            });
        });
